前言

Tinymce 是一个轻量级的、易于使用的在线编辑器，适用于各种文档类型，包括文章、博客、聊天、邮件等。以下是一些关于 Tinymce 的详细信息：

  1. 官方支持：Tinymce 是由清华KEG实验室和智谱AI公司开发的，拥有强大的技术支持。
  2. 功能强大：Tinymce 集成了多种功能，如富文本编辑、图片插入、表格操作、代码高亮、搜索和替换、插入datetime等，满足不同文档类型的编辑需求。
  3. 用户友好：Tinymce 界面简洁明了，操作简单易学，即使是非技术用户也能轻松掌握使用方法。
  4. 扩展性强：Tinymce 可以通过添加自定义插件和样式来满足项目不同的需求， also, you can use tinymce as a plugin in a more sophisticated way

安装依赖

npm i tinymce@5.1.0 -S
npm i @tinymce/tinymce-vue@3.0.1 -

环境配置

安装完依赖后在node_modules中找到tinymce，将skins复制到src/assets/tinymce下

main.js中注册

import tinymce from 'tinymce';
Vue.prototype.$tinymce = tinymce;

import request from '@/utils/request'

export function upload(file) {
  // 创建 FormData 对象
  const formData = new FormData()
  // 添加文件到 FormData 对象
  formData.append('file', file)

  // 设置请求头
  const headers = {
    'Content-Type': 'multipart/form-data; boundary=' + formData._boundary
  }

  return request({
    url: '/file/upload',
    method: 'post',
    data: formData, // 使用 FormData 对象作为请求体
    headers: headers // 应用自定义请求头
  })
}

package com.vueadmin.service.impl;


import com.vueadmin.common.Result;
import com.vueadmin.enums.AppExceptionCodeMsg;
import com.vueadmin.exception.AppException;
import com.vueadmin.service.FileService;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.core.io.FileSystemResource;
import org.springframework.core.io.Resource;
import org.springframework.http.MediaType;
import org.springframework.http.ResponseEntity;
import org.springframework.stereotype.Service;
import org.springframework.web.multipart.MultipartFile;
import ws.schild.jave.*;

import java.io.File;
import java.io.IOException;
import java.io.InputStream;
import java.nio.file.Files;
import java.nio.file.Path;
import java.nio.file.StandardCopyOption;
import java.time.LocalDateTime;
import java.time.format.DateTimeFormatter;


@Service
@Slf4j
public class FileServiceImpl implements FileService {


    @Value("${file.path}")
    private String originPath;



    @Override
    public Result<String> upload(MultipartFile file) throws IOException {
        String extName = ""; // 扩展名格式：
        if (file.getOriginalFilename().lastIndexOf(".") > 0) {
            extName = file.getOriginalFilename().substring(file.getOriginalFilename().lastIndexOf("."));
        } else {
            throw new AppException(AppExceptionCodeMsg.FILE_FORMAT_ERROR);
        }
        if (extName.equals(".mp4")) {
            return uploadVideo(file);
        } else if (extName.equals(".jpg") || extName.equals(".png")) {
            return uploadImage(file);
        } else {
            throw new AppException(AppExceptionCodeMsg.FILE_FORMAT_ERROR);
        }
    }

    @Override
    public ResponseEntity<Resource> previewImage(String imageName) throws IOException {
        // 获取图片文件路径
        String imagePath = originPath + "\\" + imageName;
        // 创建文件对象
        File imageFile = new File(imagePath);

        if (imageFile.exists()) {
            // 使用FileSystemResource创建资源对象
            Resource resource = new FileSystemResource(imageFile);
            // 设置响应的Content-Type为图片类型
            String contentType = Files.probeContentType(imageFile.toPath());
            // 返回响应实体
            return ResponseEntity.ok()
                    .contentType(MediaType.parseMediaType(contentType))
                    .body(resource);
        }
        throw new AppException(AppExceptionCodeMsg.FILE_NOT_FOUND_ERROR);
    }


    /**
     * 创建文件
     * @param extName
     * @return
     * @throws IOException
     */
    private File mkdirsmy(String extName) throws IOException {
        log.info("originPath:{}", originPath);
        // 获取当前日期时间
        LocalDateTime currentDateTime = LocalDateTime.now();
        // 创建日期时间格式化器
        DateTimeFormatter formatter = DateTimeFormatter.ofPattern("yyyyMMddHHmmssSSS");
        // 格式化输出当前日期时间
        String formattedDateTime = currentDateTime.format(formatter);
        System.out.println("当前日期时间：" + formattedDateTime);
        File file = new File(originPath, formattedDateTime + extName); //完整文件路径
        if (!file.getParentFile().exists()) {
            file.getParentFile().mkdirs();
        }
        file.createNewFile();
        return file;
    }


    /**
     * 上传图片
     *
     * @param file
     * @return
     * @throws IOException
     */
    public Result uploadVideo(MultipartFile file) throws IOException {
        String extName = ""; // 扩展名格式：
        if (file.getOriginalFilename().lastIndexOf(".") > 0) {
            extName = file.getOriginalFilename().substring(file.getOriginalFilename().lastIndexOf("."));
        } else {
            throw new AppException(AppExceptionCodeMsg.FILE_FORMAT_ERROR);
        }

        try {
            File source = mkdirsmy(extName); //获取目标文件路径
            File target = mkdirsmy(extName);
            InputStream inputStream = file.getInputStream(); // 获取源文件输入流
            Path path = source.toPath(); // 将输入流写入目标文件
            Files.copy(inputStream, path, StandardCopyOption.REPLACE_EXISTING);
            //视频格式转换
            videoConvert(source, target);
            String targetName = target.getName();
            // 文件访问路径
            String url = "http://localhost:8888/file/display/" + targetName;
            System.out.println(url);
            //删除源文件
            source.delete();
            return Result.success("上传成功", url);
        } catch (AppException e) {
            throw e;
        }
    }

    /**
     * 上传图片
     *
     * @param file
     * @return
     * @throws IOException
     */
    public Result uploadImage(MultipartFile file) throws IOException {

        String extName = ""; // 扩展名格式：
        if (file.getOriginalFilename().lastIndexOf(".") > 0) {
            extName = file.getOriginalFilename().substring(file.getOriginalFilename().lastIndexOf("."));
        } else {
            throw new AppException(AppExceptionCodeMsg.FILE_FORMAT_ERROR);
        }
        try {
            File destinationFile = mkdirsmy(extName); //获取目标文件路径
            InputStream inputStream = file.getInputStream(); // 获取源文件输入流
            Path path = destinationFile.toPath(); // 将输入流写入目标文件
            Files.copy(inputStream, path, StandardCopyOption.REPLACE_EXISTING);
            Path fileName = path.getFileName();
            // 文件访问路径
            String url = "http://localhost:8888/file/display/" + fileName;
            return Result.success("上传成功", url);
        } catch (AppException e) {
            throw e;
        }
    }

    public void videoConvert(File source, File target) {
        AudioAttributes audio = new AudioAttributes();
        audio.setCodec("libmp3lame");//音频编码格式
        //audio.setBitRate(new Integer(56000));//设置比特率，比特率越大，转出来的音频越大（默认是128000，最好默认就行，有特殊要求再设置）
        audio.setChannels(Integer.valueOf(1));
        audio.setSamplingRate(Integer.valueOf(22050));
        VideoAttributes video = new VideoAttributes();
        video.setCodec("libx264");//视屏编码格式
        //video.setBitRate(new Integer(56000));//设置比特率，比特率越大，转出来的视频越大（默认是128000,最好默认就行，有特殊要求再设置）
        video.setFrameRate(Integer.valueOf(15));//数值设置小了，视屏会卡顿
        EncodingAttributes attrs = new EncodingAttributes();
        attrs.setFormat("mp4");
        attrs.setAudioAttributes(audio);
        attrs.setVideoAttributes(video);
        Encoder encoder = new Encoder();
        MultimediaObject multimediaObject = new MultimediaObject(source);
        try {
            encoder.encode(multimediaObject, target, attrs);
        } catch (IllegalArgumentException e) {
            e.printStackTrace();
        } catch (InputFormatException e) {
            e.printStackTrace();
        } catch (EncoderException e) {
            e.printStackTrace();
        }
    }
