springboot整合knife4j

前言

Knife4j 是一款基于 Spring Boot 与 Swagger 的增强型 API 文档生成工具。它在 Swagger 的基础上进行了一些扩展和优化，提供了更丰富的功能和更友好的界面。主要特点包括：

  1. 优雅的界面: Knife4j 提供了美观、易用的界面，更加直观地展示了 API 文档信息。用户可以通过交互式界面浏览和测试 API 接口。
  2. 增强的注解支持: 支持的注解更多，可以通过一些自定义注解来扩展接口文档的描述，包括参数校验、数据格式等方面的说明。
  3. 参数测试和调试: Knife4j 允许用户在界面中直接输入参数，进行接口的测试和调试，方便快捷。
  4. 权限控制: 支持权限控制，可以对不同用户或用户组设置不同的 API 访问权限。
  5. 接口管理: 提供接口在线管理功能，包括接口的新增、删除、编辑等操作，便于团队协作与版本管理。
  6. 模块化设计: 支持接口按模块进行组织管理，使得接口文档更加清晰易读。

Knife4j 是一款非常实用的 API 文档生成工具，使得 API 接口的开发、测试和管理更加高效和便捷。

引入依赖 （springboot 2.7.5）

        <dependency>
            <groupId>com.github.xiaoymin</groupId>
            <artifactId>knife4j-openapi3-spring-boot-starter</artifactId>
            <version>4.3.0</version>
        </dependency>

配置文件

# spring-doc-openapi项目配置
spring-doc:
  swagger-ui:
    path: /swagger-ui.html
    tags-sorter: alpha
    operations-sorter: alpha
  api-docs:
    path: /v3/api-docs
  group-configs:
    - group: '默认分组'
      paths-to-match: '/**'
      packages-to-scan: com.yangpeng

# knife4j的增强配置，不需要增强可以不配
knife4j:
  enable: true
  setting:
    enable-footer: false

配置knife4j项目信息

@Configuration
public class Knife4jConfig {
    @Bean
    public OpenAPI customOpenAPI() {
        return new OpenAPI()
                .info(new Info()
                        .title("Knife4j示例项目 - 接口文档")
                        .description("项目简介，支持Markdown格式：`这里是代码标签哦`，**这里是强调哦**")
                        .version("V1.0")
                        .contact(new Contact().name("作者").email("yangpeng@qq.com"))
                );
    }
}

编写测试

package com.yangpeng.springboot_demo.controller;

import com.alibaba.fastjson.JSON;
import com.fasterxml.jackson.databind.util.JSONPObject;
import com.yangpeng.springboot_demo.entity.Emp;
import com.yangpeng.springboot_demo.sevice.EmpService;
import com.yangpeng.springboot_demo.utils.RedisUtil;
import io.swagger.v3.oas.annotations.Operation;
import io.swagger.v3.oas.annotations.Parameter;
import io.swagger.v3.oas.annotations.Parameters;
import io.swagger.v3.oas.annotations.enums.ParameterIn;
import io.swagger.v3.oas.annotations.tags.Tag;
import org.apache.tomcat.util.json.JSONParser;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.data.redis.core.RedisTemplate;
import org.springframework.lang.Nullable;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestParam;
import org.springframework.web.bind.annotation.RestController;

import java.util.List;

/**
 * @author yangpeng
 * @version 1.0.0
 * @date 2023年09月11日 16:58:01
 * @packageName com.yangpeng.datasource.controller
 * @className EmpController
 */
@RestController
@Tag(name = "应用1-Controller1")
@RequestMapping("test")
public class EmpController {

    @Autowired
    private EmpService empService;

    @Autowired
    private RedisUtil redisUtil;

    @Operation(summary = "查询数据库数据")
    @GetMapping(value = "/list")
    public List<Emp> showEmpList() {
        List<Emp> list = empService.list();

        String jsonString = JSON.toJSONString(list);
        redisUtil.set("empList", jsonString);

        return list;
    }

    @Operation(summary = "查验应用是否存活")
    @GetMapping(value = "/isAlive")
    public String isAlive() {
        return "hello";

    }

    @Operation(summary = "查验Redis是否存活")
    @GetMapping(value = "/testRedis")
    public String testRedis() {
        Object name = redisUtil.get("empList");
        return name.toString();

    }

    @Operation(summary = "向客人问好")
    @Parameter(name = "name", description = "姓名", in = ParameterIn.QUERY)
    @GetMapping("/hello")
    public String sayHi(@RequestParam String name) {
        return "Hi:" + name;
    }


    @Operation(summary = "测试参数可以为空")
    @Parameters({
            @Parameter(name = "name", description = "姓名", in = ParameterIn.QUERY)
    })
    @GetMapping("/nullable")
    public String testNullable(@Nullable String name) {
        return "Hi:" + name;
    }


}
